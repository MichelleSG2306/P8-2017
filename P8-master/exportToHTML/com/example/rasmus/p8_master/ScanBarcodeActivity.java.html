43       private void createCameraSource() { 
44    
45           BarcodeDetector barcodeDetector = new BarcodeDetector.Builder(this).build(); 
46           final CameraSource cameraSource = new CameraSource.Builder(this, barcodeDetector) 
47                   .setAutoFocusEnabled(true) 
48                   .setRequestedPreviewSize(1600, 1024) 
49                   .build(); 
50    
51           cameraPreview.getHolder().addCallback(new SurfaceHolder.Callback() { 
52               @Override 
53               public void surfaceCreated(SurfaceHolder holder) { 
54                   if (ActivityCompat.checkSelfPermission(ScanBarcodeActivity.this, Manifest.permission.CAMERA) != PackageManager.PERMISSION_GRANTED) { 
55                       // TODO: Consider calling 
56                       //    ActivityCompat#requestPermissions 
57                       // here to request the missing permissions, and then overriding 
58                       //   public void onRequestPermissionsResult(int requestCode, String[] permissions, 
59                       //                                          int[] grantResults) 
60                       // to handle the case where the user grants the permission. See the documentation 
61                       // for ActivityCompat#requestPermissions for more details. 
62                       return; 
63                   } 
64                   try { 
65                       cameraSource.start(cameraPreview.getHolder()); 
66                   } catch (IOException e) { 
67                       e.printStackTrace(); 
68                   } 
69               } 
70    
71               @Override 
72               public void surfaceChanged(SurfaceHolder holder, int format, int width, int height) { 
73    
74               } 
75    
76               @Override 
77               public void surfaceDestroyed(SurfaceHolder holder) { 
78                   cameraSource.stop(); 
79               } 
80           }); 
81    
82           barcodeDetector.setProcessor(new Detector.Processor<Barcode>() { 
83               @Override 
84               public void release() { 
85    
86               } 
87    
88               @Override 
89               //This is an array for barcodes 
90               public void receiveDetections(Detector.Detections<Barcode> detections) { 
91                   final SparseArray<Barcode> barcodes = detections.getDetectedItems(); 
92                   if (barcodes.size()>0) { 
93                       Intent intent = new Intent(); 
94                       intent.putExtra("barcode", barcodes.valueAt(0)); 
95                       //This only gets the latest barcode in the array 
96                       setResult(CommonStatusCodes.SUCCESS, intent); 
97                       finish(); 
98                   } 
99               } 
100          }); 
101      }
