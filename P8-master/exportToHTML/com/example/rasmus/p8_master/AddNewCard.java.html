31       public void scanBarcode(View v) { 
32    
33           Intent intent = new Intent(this, ScanBarcodeActivity.class); 
34           startActivityForResult(intent, 0); 
35    
36           this.requestCameraPermission(); 
37       }

39       public void requestCameraPermission() { 
40           if (ContextCompat.checkSelfPermission(this, Manifest.permission.CAMERA) 
41                   != PackageManager.PERMISSION_GRANTED) { 
42               if (ActivityCompat.shouldShowRequestPermissionRationale(this, 
43                       Manifest.permission.CAMERA)) { 
44               } else { 
45                   ActivityCompat.requestPermissions(this, 
46                           new String[]{Manifest.permission.CAMERA},0); 
47               } 
48    
49           } 
50       } 
51    
52       public void onActivityResult(int requestCode, int resultCode, Intent data) { 
53           if (requestCode == 0) { 
54               if (resultCode == CommonStatusCodes.SUCCESS) { 
55                   if (data != null) { 
56                       Barcode barcode = data.getParcelableExtra("barcode"); 
57                       scannedBarcode = barcode.rawValue; 
58    
59                       EditText showBarcode = (EditText) findViewById(R.id.barcodeInput); 
60                       showBarcode.setText(scannedBarcode, TextView.BufferType.EDITABLE); 
61    
62                   } 
63               } 
64    
65           } else { 
66    
67               super.onActivityResult(requestCode, resultCode, data); 
68           } 
69    
70       } 
71    
72       public void saveCard (View v){ 
73           EditText nameInput = (EditText)findViewById(R.id.nameInput); 
74           String nameOfCardInput = nameInput.getText().toString(); 
75    
76           EditText barcodeInput = (EditText) findViewById(R.id.barcodeInput); 
77           String barcode = barcodeInput.getText().toString(); 
78    
79           if(TextUtils.isEmpty(nameOfCardInput) || TextUtils.isEmpty(barcode)) 
80           { 
81    
82               TextView infoMissing = (TextView) findViewById(R.id.missingBarcodeName); 
83               infoMissing.setVisibility(View.VISIBLE); 
84    
85           } 
86    
87           else { 
88    
89               blueOrGreenCard = R.drawable.othercardb; 
90    
91               DBHelper dbHelper = new DBHelper(AddNewCard.this); 
92               dbHelper.insertCard(nameOfCardInput, blueOrGreenCard, barcode); 
93               Toast.makeText(AddNewCard.this, "Card added", Toast.LENGTH_SHORT).show(); 
94    
95               Intent intent = new Intent(getApplicationContext(), MainActivity.class); 
96               startActivity(intent); 
97           } 
98       }
